rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // PDF files - everyone can read, only authenticated users can write/delete
    // (We'll handle admin check in the application code since Storage rules 
    // can't easily access Realtime Database)
    match /pdfs/{allPaths=**} {
      allow read: if true;  // Public read access for all users
      allow write, delete: if isAuthenticated();
    }
    
    // PowerPoint files - everyone can read, only authenticated users can write/delete
    match /pptx/{allPaths=**} {
      allow read: if true;  // Public read access for all users
      allow write, delete: if isAuthenticated();
    }
    
    // Public materials - everyone can read, only authenticated users can write/delete
    match /public_materials/{allPaths=**} {
      allow read: if true;  // Public read access for all users
      allow write, delete: if isAuthenticated();
    }
    
    // Admin-only folders
    match /admin/{allPaths=**} {
      allow read, write, delete: if isAuthenticated();
    }
    
    // Deny all other paths by default
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
